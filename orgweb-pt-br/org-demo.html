* Verificar a taxa de minificação do CSS

Babel

Faça programação literária no Org, com execução de código ao vivo, semelhante a um notebook, no buffer. Código 
#+begin_src python
 
from pathlib import Path
cssRatios = []
for css_min in Path("resources/style").glob("*.min.css"):
    css = css_min.with_suffix('').with_suffix('.css')
    cssRatios.append([css.name,
    "{:.0f}% minificado ({:4.1f} KiB)".format( 100 *
                      css_min.stat().st_size / css.stat().st_size,
                      css_min.stat().st_size / 1000)])
return cssRatios

#+end_src
